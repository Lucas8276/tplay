package lucas.entities;

public class PieceSquare {
    private char[][] square;
    private int value;
    public  PieceSquare() {
        square= new char[4][4];
        // Dibuja la pieza en forma de "Square"
        square[0][0] = '.';
        square[1][0] = '.';
        square[0][1] = '.';
        square[1][1] = '.';
        value= 0;  
}
public char[][]getPieceSquare(){
    return square;
}
public int getValue(){
    return value;
}
